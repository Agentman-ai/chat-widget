  /* GPU acceleration for smooth animations */
  transform: translateZ(0);
  will-change: transform, box-shadow;
  backface-visibility: hidden;
}

/* 2. FOCUS STATE WITH EXPANSION */
.daydream-input-container:has(:focus-within) {
  min-height: 88px; /* Double height on focus */
  height: auto;
  /* Blue glow shadow */
  box-shadow: 0px 0px 32px rgba(201, 208, 245, 0.8);
}

/* 3. DYNAMIC TEXTAREA POSITIONING */
.daydream-textarea {
  field-sizing: content; /* Modern auto-sizing */
  position: absolute; /* Centered when collapsed */
  transition: all 0.3s ease;
}

.daydream-input-container:has(:focus-within) .daydream-textarea {
  position: relative; /* Normal flow when expanded */
  height: auto;
  min-height: 32px;
}

/* 4. PROGRESSIVE BUTTON DISCLOSURE */
.daydream-buttons-wrapper {
  position: absolute;
  bottom: 8px;
  opacity: 0; /* Hidden by default */
  pointer-events: none;
  transition: opacity 0.3s ease;
}

.daydream-input-container:has(:focus-within) .daydream-buttons-wrapper {
  opacity: 1; /* Show on focus */
